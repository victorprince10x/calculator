<script>
const MONTHS       = ['January','February','March','April','May','June','July','August','September','October','November','December'];
const MONTHS_SHORT = ['JAN','FEB','MAR','APR','MAY','JUN','JUL','AUG','SEP','OCT','NOV','DEC'];

let esiids = [];
let activeTab = 0;

function createEsiid(name='', esiidNum='', address='') {
  return { id: Date.now(), name, esiidNum, address, usage: new Array(12).fill(0) };
}

function addEsiid() {
  const idx = esiids.length;
  esiids.push(createEsiid(`ESIID ${idx + 1}`));
  activeTab = idx;
  renderAll();
  calculate();
}

function removeEsiid(idx) {
  if (esiids.length <= 1) return;
  esiids.splice(idx, 1);
  if (activeTab >= esiids.length) activeTab = esiids.length - 1;
  renderAll();
  calculate();
}

function switchTab(idx) {
  activeTab = idx;
  renderTabs();
  renderPanels();
}

function saveUsageFromDOM() {
  esiids.forEach((e, idx) => {
    MONTHS.forEach((_, m) => {
      const el = document.getElementById(`u_${idx}_${m}`);
      if (el) e.usage[m] = parseFloat(el.value) || 0;
    });
  });
}

let dragSrcIdx = null;

function renderTabs() {
  const row = document.getElementById('esiidTabsRow');
  const addBtn = row.querySelector('.add-esiid-tab-btn');
  Array.from(row.querySelectorAll('.esiid-tab')).forEach(t => t.remove());

  esiids.forEach((e, idx) => {
    const total = e.usage.reduce((a,b) => a+b, 0);
    const tab = document.createElement('div');
    tab.className = 'esiid-tab' + (idx === activeTab ? ' active' : '');
    tab.draggable = true;
    tab.dataset.idx = idx;
    tab.innerHTML = `
      <span class="esiid-tab-name" onclick="switchTab(${idx})">${e.name || 'ESIID ' + (idx+1)}</span>
      <span class="esiid-tab-total" onclick="switchTab(${idx})">${total > 0 ? fmtNum(total)+' kWh' : 'empty'}</span>
      ${esiids.length > 1 ? `<button class="tab-remove" onclick="removeEsiid(${idx})" title="Remove">✕</button>` : ''}
    `;
    tab.addEventListener('dragstart', (e) => {
      dragSrcIdx = idx;
      setTimeout(() => tab.classList.add('dragging'), 0);
    });
    tab.addEventListener('dragend', () => {
      tab.classList.remove('dragging');
      document.querySelectorAll('.esiid-tab').forEach(t => t.classList.remove('drag-over'));
    });
    tab.addEventListener('dragover', (e) => {
      e.preventDefault();
      document.querySelectorAll('.esiid-tab').forEach(t => t.classList.remove('drag-over'));
      tab.classList.add('drag-over');
    });
    tab.addEventListener('drop', (e) => {
      e.preventDefault();
      const destIdx = parseInt(tab.dataset.idx);
      if (dragSrcIdx === null || dragSrcIdx === destIdx) return;
      const moved = esiids.splice(dragSrcIdx, 1)[0];
      esiids.splice(destIdx, 0, moved);
      activeTab = destIdx;
      dragSrcIdx = null;
      renderAll();
      calculate();
    });
    row.insertBefore(tab, addBtn);
  });
}

function renderPanels() {
  const container = document.getElementById('esiidPanels');
  container.innerHTML = '';
  esiids.forEach((e, idx) => {
    const panel = document.createElement('div');
    panel.className = 'esiid-panel' + (idx === activeTab ? ' active' : '');

    let usageInputs = '';
    MONTHS.forEach((m, mi) => {

    });

    const total = e.usage.reduce((a,b) => a+b, 0);

    panel.innerHTML = `
      <div class="esiid-meta">
        <div class="esiid-meta-field">
          <label>ESIID Number</label>
          <input type="text" id="esiidNum_${idx}" value="${e.esiidNum}" placeholder="Enter ESIID Number" oninput="onEsiidNumInput(${idx}, this.value)" onpaste="onEsiidNumPaste(${idx}, event)">

        </div>

function onUsageInput(idx, month, val) {
  esiids[idx].usage[month] = parseFloat(val) || 0;
  const total = esiids[idx].usage.reduce((a,b) => a+b, 0);
  const subEl = document.getElementById(`sub_${idx}`);
  if (subEl) subEl.textContent = fmtNum(total);
  const tabs = document.querySelectorAll('.esiid-tab');
  if (tabs[idx]) {
    const totEl = tabs[idx].querySelector('.esiid-tab-total');
    if (totEl) totEl.textContent = total > 0 ? fmtNum(total)+' kWh' : 'empty';
  }
  calculate();
}

function onEsiidNumInput(idx, val) {
  esiids[idx].esiidNum = val;
  if (!esiids[idx].customLabel) {
    esiids[idx].name = val || ('ESIID ' + (idx + 1));
  }
  renderTabs();
}

function onEsiidNumPaste(idx, e) {
  const text = e.clipboardData && e.clipboardData.getData('text') || '';
  const entries = text
    .split(/[\n\r\s,;]+/)
    .map(s => s.trim())
    .filter(s => s.length > 0);

  if (entries.length <= 1) return;
  e.preventDefault();

  esiids[idx].esiidNum = entries[0];
  if (!esiids[idx].customLabel) esiids[idx].name = entries[0];

  entries.slice(1).forEach((num) => {
    const newE = createEsiid(num, num);
    newE.esiidNum = num;
    newE.name = num;
    esiids.push(newE);
  });

  activeTab = idx;
  renderAll();
  calculate();

  const firstEl = document.getElementById('esiidNum_' + idx);
  if (firstEl) firstEl.value = entries[0];
  entries.slice(1).forEach((num, i) => {
    const el = document.getElementById('esiidNum_' + (idx + 1 + i));
    if (el) el.value = num;
  });
}

function onEsiidLabelInput(idx, val) {
  esiids[idx].customLabel = val.length > 0;
  esiids[idx].name = val || esiids[idx].esiidNum || ('ESIID ' + (idx + 1));
  renderTabs();
}

function clearCurrentEsiid() {
  MONTHS.forEach((_, m) => {
    esiids[activeTab].usage[m] = 0;
    const el = document.getElementById(`u_${activeTab}_${m}`);
    if (el) el.value = '';
  });
  const subEl = document.getElementById(`sub_${activeTab}`);
  if (subEl) subEl.textContent = '0';
  calculate();
}

function onPasteUsage(e, idx) {
  setTimeout(() => {
    const raw = document.getElementById(`paste_${idx}`).value;
    const status = document.getElementById(`pasteStatus_${idx}`);

    const nums = raw
      .split(/[\n\r\t,; ]+/)
      .map(s => parseFloat(s.replace(/,/g, '')))
      .filter(n => !isNaN(n) && n >= 0);

    if (nums.length === 0) {
      status.textContent = '';
      status.className = 'paste-status';
      return;
    }

    if (nums.length < 12) {
      status.textContent = `⚠ Found ${nums.length} value${nums.length>1?'s':''} — need 12 (January through December)`;
      status.className = 'paste-status err';
      return;
    }

    if (nums.length > 12) {
      status.textContent = `⚠ Found ${nums.length} values — using first 12`;
      status.className = 'paste-status err';
    }

    const toUse = nums.slice(0, 12);
    toUse.forEach((val, m) => {
      esiids[idx].usage[m] = val;
      const el = document.getElementById(`u_${idx}_${m}`);
      if (el) el.value = val > 0 ? val : '';
    });

    const total = toUse.reduce((a,b) => a+b, 0);
    const subEl = document.getElementById(`sub_${idx}`);
    if (subEl) subEl.textContent = fmtNum(total);

    if (nums.length === 12) {
      status.textContent = `✓ All 12 months loaded · ${fmtNum(total)} kWh total`;
      status.className = 'paste-status ok';
    }

    renderTabs();
    calculate();
  }, 0);
}

function fmt(n, dec=2) { return '$' + n.toFixed(dec).replace(/\B(?=(\d{3})+(?!\d))/g, ','); }
function fmtNum(n) { return Math.round(n).toString().replace(/\B(?=(\d{3})+(?!\d))/g, ','); }
function fmtPct(n) { return n.toFixed(1) + '%'; }

function calculate() {
  const cr = parseFloat(document.getElementById('currentRate').value) || 0;
  const termRates = {
    12: parseFloat(document.getElementById('rate12').value) || 0,
    24: parseFloat(document.getElementById('rate24').value) || 0,
    36: parseFloat(document.getElementById('rate36').value) || 0,
    48: parseFloat(document.getElementById('rate48').value) || 0,
    60: parseFloat(document.getElementById('rate60').value) || 0,
  };

  const nr   = termRates[12];
  const diff = cr - nr;

  const combined = new Array(12).fill(0);
  esiids.forEach((e) => {
    e.usage.forEach((u, m) => { combined[m] += u; });
  });

  const annualUsage = combined.reduce((a,b) => a+b, 0);
  const maxUsage    = Math.max(...combined, 1);

  let totalCurrent=0, totalNew=0, totalSavings=0;
  const monthlySavings=[];
  const tbody = document.getElementById('monthTable');
  tbody.innerHTML = '';

  combined.forEach((u, i) => {
    const cur = u * cr, nw = u * nr, sav = u * diff;
    totalCurrent += cur; totalNew += nw; totalSavings += sav;
    monthlySavings.push(sav);
    const pct = cur > 0 ? (sav/cur)*100 : 0;
    const barPct = (u/maxUsage)*100;
    tbody.innerHTML += `
      <tr>
        <td>${MONTHS[i]}</td>
        <td>${u>0?fmtNum(u):'—'}</td>
        <td>${u>0?fmt(cur):'—'}</td>
        <td>${u>0?fmt(nw):'—'}</td>
        <td class="savings-cell">${u>0?fmt(sav):'—'}</td>
        <td>${u>0?fmtPct(pct):'—'}</td>
        <td class="bar-cell"><div class="bar-track"><div class="bar-fill" style="width:${barPct}%"></div></div></td>
      </tr>`;
  });

  const overallPct = totalCurrent > 0 ? (totalSavings/totalCurrent)*100 : 0;
  document.getElementById('monthFoot').innerHTML = `
    <tr>
      <td>ANNUAL TOTAL</td><td>${fmtNum(annualUsage)}</td>
      <td>${fmt(totalCurrent)}</td><td>${fmt(totalNew)}</td>
      <td class="savings-cell">${fmt(totalSavings)}</td>
      <td>${fmtPct(overallPct)}</td><td class="bar-cell"></td>
    </tr>`;

  const peakIdx = monthlySavings.indexOf(Math.max(...monthlySavings));
  document.getElementById('kpiGrid').innerHTML = `
    <div class="kpi-card highlight">
      <div class="kpi-label">Annual Savings (12-Mo Rate)</div>
      <div class="kpi-value">${fmt(totalSavings,0)}</div>
      <div class="kpi-sub">${fmtPct(overallPct)} reduction in energy costs</div>
    </div>
    <div class="kpi-card">
      <div class="kpi-label">Monthly Average</div>
      <div class="kpi-value">${fmt(totalSavings/12,0)}</div>
      <div class="kpi-sub">Per month avg savings</div>
    </div>
    <div class="kpi-card">
      <div class="kpi-label">Total Usage</div>
      <div class="kpi-value">${fmtNum(annualUsage)}</div>
      <div class="kpi-sub">kWh annual consumption</div>
    </div>
    <div class="kpi-card">
      <div class="kpi-label">Rate Reduction (12-Mo)</div>
      <div class="kpi-value">${fmtPct(cr>0?(diff/cr)*100:0)}</div>
      <div class="kpi-sub">$${diff.toFixed(4)}/kWh price drop</div>
    </div>
    <div class="kpi-card">
      <div class="kpi-label">Peak Savings Month</div>
      <div class="kpi-value">${MONTHS[peakIdx].slice(0,3).toUpperCase()}</div>
      <div class="kpi-sub">${fmt(monthlySavings[peakIdx])} in ${MONTHS[peakIdx]}</div>
    </div>`;

  const chartBars = document.getElementById('chartBars');
  chartBars.innerHTML = '';
  const nonZero = monthlySavings.filter(s => s > 0);
  if (nonZero.length > 0) {
    const maxSav = Math.max(...monthlySavings);
    const minSav = Math.min(...nonZero);
    const baseline = minSav * 0.8;
    const range = Math.max(maxSav - baseline, 1);
    monthlySavings.forEach((s,i) => {
      const pct = s > 0 ? 15 + ((s - baseline) / range) * 82 : 2;
      chartBars.innerHTML += `
        <div class="chart-col">
          <div class="chart-amount">${s>0?'$'+Math.round(s).toLocaleString():''}</div>
          <div class="chart-bar" style="height:${pct}%" title="${MONTHS[i]}: ${fmt(s)}"></div>
          <div class="chart-month">${MONTHS_SHORT[i]}</div>
        </div>`;
    });
  }

  const terms = [
    { months: 12,  rate: termRates[12]  },
    { months: 24,  rate: termRates[24]  },
    { months: 36,  rate: termRates[36]  },
    { months: 48,  rate: termRates[48]  },
    { months: 60,  rate: termRates[60]  },
  ];

  const mygrid=document.getElementById('multiyearGrid'), mytable=document.getElementById('multiyearTable');
  mygrid.innerHTML=''; mytable.innerHTML='';

  terms.forEach(({ months, rate }, i) => {
    const years = months / 12;
    const termDiff = cr - rate;
    const annualSav = annualUsage * termDiff;
    const annualCur = annualUsage * cr;
    const annualNew = annualUsage * rate;
    const cumCur = annualCur * years;
    const cumNew = annualNew * years;
    const cumSav = annualSav * years;
    const pct = cumCur > 0 ? (cumSav / cumCur) * 100 : 0;
    const isFirst = i === 0;
    mygrid.innerHTML += `
      <div class="year-card ${isFirst?'active':''}">
        <div class="year-card-label">${months}-Month Term</div>
        <div class="year-card-savings">${fmt(cumSav,0)}</div>
        <div class="year-card-pct">${fmtPct(pct)} savings · $${rate.toFixed(4)}/kWh</div>
      </div>`;
    mytable.innerHTML += `
      <tr>
        <td style="text-align:left;font-family:'DM Sans',sans-serif;color:var(--text-muted)">${months}-Month Term</td>
        <td>${months}</td><td>${fmt(cumCur)}</td><td>${fmt(cumNew)}</td>
        <td class="savings-cell">${fmt(cumSav)}</td><td>${fmtPct(pct)}</td>
      </tr>`;
  });

  const banner = document.getElementById('combinedBanner');
  if (esiids.length > 1) {
    banner.style.display = 'flex';
    document.getElementById('combinedCount').textContent = esiids.length;
    document.getElementById('combinedKwh').textContent = fmtNum(annualUsage);
    document.getElementById('combinedSavings').textContent = fmt(totalSavings, 0);
    document.getElementById('combined5yr').textContent = fmt(totalSavings * 5, 0);
  } else {
    banner.style.display = 'none';
  }

  const rateRedPct = cr > 0 ? (diff/cr)*100 : 0;
  const rrVal = document.getElementById('printRateReductionVal');
  const rrSub = document.getElementById('printRateReductionSub');
  if (rrVal) rrVal.textContent = fmtPct(rateRedPct);
  if (rrSub) rrSub.textContent = '$' + diff.toFixed(4) + '/kWh price drop';

  document.getElementById('genDate').textContent = new Date().toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'});
}

function saveSession() {
  const data = {
    prospectName: document.getElementById('prospectName').value,
    currentRate: document.getElementById('currentRate').value,
    rate12: document.getElementById('rate12').value,
    rate24: document.getElementById('rate24').value,
    rate36: document.getElementById('rate36').value,
    rate48: document.getElementById('rate48').value,
    rate60: document.getElementById('rate60').value,
    esiids: esiids
  };
  localStorage.setItem('energySaverSession', JSON.stringify(data));
  const btn = document.querySelector('[onclick="saveSession()"]');
  const orig = btn.textContent;
  btn.textContent = '✓ Saved!';
  btn.style.color = 'var(--gold)';
  setTimeout(() => { btn.textContent = orig; btn.style.color = ''; }, 2000);
}

function loadSession() {
  const raw = localStorage.getItem('energySaverSession');
  if (!raw) return;
  const data = JSON.parse(raw);
  document.getElementById('prospectName').value = data.prospectName || '';
  document.getElementById('currentRate').value = data.currentRate || '';
  document.getElementById('rate12').value = data.rate12 || '';
  document.getElementById('rate24').value = data.rate24 || '';
  document.getElementById('rate36').value = data.rate36 || '';
  document.getElementById('rate48').value = data.rate48 || '';
  document.getElementById('rate60').value = data.rate60 || '';
  esiids = data.esiids || [createEsiid('ESIID 1')];
  activeTab = 0;
  renderAll();
  calculate();
}

function checkSavedSession() {

  }
}
